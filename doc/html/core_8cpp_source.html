342 struct boot_flags bootmode_detect(void) {
343  int reset_reason, bootmode;
344  asm (
345  "movi %0, 0x60000600\n\t"
346  "movi %1, 0x60000200\n\t"
347  "l32i %0, %0, 0x114\n\t"
348  "l32i %1, %1, 0x118\n\t"
349  : "+r" (reset_reason), "+r" (bootmode) /* Outputs */
350  : /* Inputs (none) */
351  : "memory" /* Clobbered */
352  );
353 
354  struct boot_flags flags;
355  flags.raw_rst_cause = (reset_reason&0xF);
356  flags.raw_bootdevice = ((bootmode>>0x10)&0x7);
357  flags.raw_bootmode = ((bootmode>>0x1D)&0x7);
358  flags.rst_normal_boot = flags.raw_rst_cause == 0x1;
359  flags.rst_reset_pin = flags.raw_rst_cause == 0x2;
360  flags.rst_watchdog = flags.raw_rst_cause == 0x4;
361  flags.bootdevice_ram = flags.raw_bootdevice == 0x1;
362  flags.bootdevice_flash = flags.raw_bootdevice == 0x3;
363  return flags;
364 }
