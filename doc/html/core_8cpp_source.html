372 struct boot_flags bootmode_detect(void) {
373  int reset_reason, bootmode;
374  asm (
375  "movi %0, 0x60000600\n\t"
376  "movi %1, 0x60000200\n\t"
377  "l32i %0, %0, 0x114\n\t"
378  "l32i %1, %1, 0x118\n\t"
379  : "+r" (reset_reason), "+r" (bootmode) /* Outputs */
380  : /* Inputs (none) */
381  : "memory" /* Clobbered */
382  );
383 
384  struct boot_flags flags;
385  flags.raw_rst_cause = (reset_reason&0xF);
386  flags.raw_bootdevice = ((bootmode>>0x10)&0x7);
387  flags.raw_bootmode = ((bootmode>>0x1D)&0x7);
388  flags.rst_normal_boot = flags.raw_rst_cause == 0x1;
389  flags.rst_reset_pin = flags.raw_rst_cause == 0x2;
390  flags.rst_watchdog = flags.raw_rst_cause == 0x4;
391  flags.bootdevice_ram = flags.raw_bootdevice == 0x1;
392  flags.bootdevice_flash = flags.raw_bootdevice == 0x3;
393  return flags;
394 }
